Method:magix.execute

inspect=>@"Loops through a list of
messages to show to end user, removing
them as it proceeds. Continues until it
has gone through all the items, and removed
them all."

Data
  msg1
    message=>Howdy 1...
  msg2
    message=>Howdy 2...
  msg3
    message=>Howdy 3...
while=>[Data].Count!=0
  set=>[.ip][..][magix.viewport.show-message][message].Value
    value=>[Data][0][message].Value
  remove=>[Data][0]
  magix.viewport.show-message
    message
